ServantActivator

ServantActivator interface

// Called when a servant is needed
incarnate(oid: Uint8Array, adapter: POA): Promise

etherealize(oid: Uint8Array, adapter: POA, serv: Servant, cleanup_in_progress: boolean, remaining_activations: boolean): Promise

ServantLocator

ServantLocator interface

// Called when a servant is needed for a request
preinvoke(oid: Uint8Array, adapter: POA, operation: string): Promise

postinvoke(oid: Uint8Array, adapter: POA, operation: string, cookie: unknown, servant: Servant): Promise

POA interface and implementation

// Create a child POA

get_servant(): Promise

set_servant(servant: Servant): Promise

activate_object(servant: Servant): Promise
// Deactivate an object
deactivate_object(oid: Uint8Array): Promise

create_reference(intf: string): Object

servant_to_id(servant: Servant): Promise
// Get the reference for a servant
servant_to_reference(servant: Servant): Promise
