Quick Example

import { init, string_to_object } from "jsr:@eai/corba";

// Initialize the ORB
await init();

// Get an object reference
const obj = string_to_object("corbaloc::localhost:2809/NameService");

// Use the object
const result = await obj._request("resolve")
